use Livewire\Attributes\Cache;
use Livewire\Attributes\Lazy;

{
    // Propri√©t√©s avec optimisations
    public $title = '';
    public $description = '';
    public $editingTask = null;
    public $search = '';
    public $filter = 'all';

    // R√®gles de validation (inchang√©es)
    protected $rules = [

    // Optimisation : Lazy loading des statistiques
    #[Lazy]
    #[Cache(seconds: 300)] // Cache 5 minutes
    public function getStatsProperty()
    {
        return cache()->remember('task_stats', 300, function () {
            $tasks = $this->tasks;

            return [
                'total' => $tasks->count(),
                'completed' => $tasks->where('completed', true)->count(),
                'pending' => $tasks->where('completed', false)->count(),
                'completion_rate' => $tasks->count() > 0
                    ? round(($tasks->where('completed', true)->count() / $tasks->count()) * 100)
                    : 0
            ];
        });
    }

    // Optimisation : Requ√™te optimis√©e avec pagination
    public function getTasksProperty()
    {
        return Task::query()
            ->select(['id', 'title', 'description', 'completed', 'created_at']) // Seulement les champs n√©cessaires
            ->when($this->search, fn($q) => $q->where('title', 'ilike', '%' . $this->search . '%'))
            ->when($this->filter === 'completed', fn($q) => $q->where('completed', true))
            ->when($this->filter === 'pending', fn($q) => $q->where('completed', false))
            ->latest()
            ->limit(100) // Limitation pour √©viter la surcharge
            ->get();
    }

    // M√©thodes avec logs pour monitoring
    public function addTask()
    {

                'title' => $this->title,

            session()->flash('message', 'T√¢che ajout√©e avec succ√®s !');

            // Invalider le cache des stats
            cache()->forget('task_stats');

            session()->flash('error', 'Erreur lors de la cr√©ation de la t√¢che.');
        }
    }

    // Autres m√©thodes avec gestion d'erreurs similaire...
    public function toggleTask($taskId)
    {

            $task->update(['completed' => !$task->completed]);

            logger('Task toggled', ['task_id' => $taskId, 'completed' => $task->completed]);
            session()->flash('message', 'T√¢che mise √† jour !');

    public function handle(Request $request, Closure $next)

                'url' => $request->fullUrl(),

                'user_agent' => $request->userAgent(),
                'ip' => $request->ip(),
            ]);
        }

        // Ajouter les headers de performance
        $response->headers->set('X-Execution-Time', round($executionTime, 2) . 'ms');
        $response->headers->set('X-Memory-Usage', round($memoryUsage / 1024 / 1024, 2) . 'MB');

        return $response;
    }

<?php

namespace App\Exceptions;

use Illuminate\Foundation\Exceptions\Handler as ExceptionHandler;
use Illuminate\Support\Facades\Log;
use Throwable;

class Handler extends ExceptionHandler
{
    protected $dontReport = [
        // Exceptions √† ne pas reporter
    ];

    protected $dontFlash = [
        'current_password',
        'password',
        'password_confirmation',
    ];

    public function register(): void
    {
        $this->reportable(function (Throwable $e) {
            // Reporting personnalis√© pour la production
            if (app()->environment('production')) {
                $this->reportProductionError($e);
            }
        });
    }

    private function reportProductionError(Throwable $e): void
    {
        $context = [
            'exception' => get_class($e),
            'message' => $e->getMessage(),
            'file' => $e->getFile(),
            'line' => $e->getLine(),
            'url' => request()->fullUrl(),
            'method' => request()->method(),
            'ip' => request()->ip(),
            'user_agent' => request()->userAgent(),
            'user_id' => auth()->id(),
            'stack_trace' => $e->getTraceAsString(),
        ];

        // Log selon la criticit√©
        if ($this->isCriticalError($e)) {
            Log::channel('critical')->critical('Erreur critique d√©tect√©e', $context);

            // Optionnel : Notification Slack
            if (config('logging.channels.slack.url')) {
                Log::channel('slack')->critical('üö® Erreur critique sur TALL Tasks', $context);
            }
        } else {
            Log::channel('daily')->error('Erreur applicative', $context);
        }
    }

    private function isCriticalError(Throwable $e): bool
    {
        $criticalExceptions = [
            \Illuminate\Database\QueryException::class,
            \PDOException::class,
            \Symfony\Component\HttpKernel\Exception\ServiceUnavailableHttpException::class,
        ];

        return in_array(get_class($e), $criticalExceptions) ||
               str_contains($e->getMessage(), 'SQLSTATE') ||
               str_contains($e->getMessage(), 'Connection refused');
    }

    public function render($request, Throwable $e)
    {
        // Pages d'erreur personnalis√©es pour la production
        if (app()->environment('production')) {
            if ($this->isHttpException($e)) {
                $statusCode = $e->getStatusCode();

                if (view()->exists("errors.{$statusCode}")) {
                    return response()->view("errors.{$statusCode}", [], $statusCode);
                }
            }
        }

        return parent::render($request, $e);
    }
}
