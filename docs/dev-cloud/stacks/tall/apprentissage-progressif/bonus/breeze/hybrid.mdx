<?php

namespace App\Livewire\Auth;

use App\Models\User;
use Illuminate\Auth\Events\Registered;
use Illuminate\Support\Facades\Auth;
use Illuminate\Support\Facades\Hash;
use Livewire\Attributes\Layout;
use Livewire\Attributes\Validate;
use Livewire\Component;

class Register extends Component
{
    #[Validate('required|string|max:255')]
    public string $name = '';

    #[Validate('required|string|email|max:255|unique:users')]
    public string $email = '';

    #[Validate('required|string|confirmed|min:8')]
    public string $password = '';

    #[Validate('required|string')]
    public string $password_confirmation = '';

    // Propri√©t√©s pour l'UX am√©lior√©e
    public bool $isLoading = false;
    public string $message = '';
    public bool $termsAccepted = false;

    // Validation en temps r√©el
    public function updated($propertyName)
    {
        $this->validateOnly($propertyName);

        // V√©rification de la force du mot de passe
        if ($propertyName === 'password') {
            $this->checkPasswordStrength();
        }
    }

    public function register(): void
    {
        $this->isLoading = true;

        // Validation des conditions d'utilisation
        if (!$this->termsAccepted) {
            $this->addError('termsAccepted', 'Vous devez accepter les conditions d\'utilisation.');
            $this->isLoading = false;
            return;
        }

        // Validation compl√®te
        $this->validate();

        // Cr√©ation de l'utilisateur
        $user = User::create([
            'name' => $this->name,
            'email' => $this->email,
            'password' => Hash::make($this->password),
        ]);

        // √âv√©nement d'inscription
        event(new Registered($user));

        // Connexion automatique
        Auth::login($user);

        // Log de l'activit√©
        logger('User registered', [
            'user_id' => $user->id,
            'email' => $user->email,
            'ip' => request()->ip()
        ]);

        // Message de succ√®s
        $this->message = 'Inscription r√©ussie ! Redirection vers le tableau de bord...';

        // Redirection
        $this->js("setTimeout(() => window.location.href = '/dashboard', 1500)");
    }

    /**
     * V√©rification de la force du mot de passe
     */
    private function checkPasswordStrength(): void
    {
        if (strlen($this->password) < 8) return;

        $strength = 0;
        if (preg_match('/[a-z]/', $this->password)) $strength++;
        if (preg_match('/[A-Z]/', $this->password)) $strength++;
        if (preg_match('/[0-9]/', $this->password)) $strength++;
        if (preg_match('/[^a-zA-Z0-9]/', $this->password)) $strength++;

        // Dispatch d'un √©v√©nement pour mettre √† jour l'indicateur
        $this->dispatch('password-strength-updated', strength: $strength);
    }

    #[Layout('layouts.guest')]
    public function render()
    {
        return view('livewire.auth.register');
    }
}
